CREATE TABLE Wishes (
    WishID INT AUTO_INCREMENT PRIMARY KEY,
    WishText VARCHAR(255) NOT NULL
);

CREATE TABLE UserWishes (
    UserID INT,
    WishID INT,
    FOREIGN KEY (UserID) REFERENCES Users(UserID),
    FOREIGN KEY (WishID) REFERENCES Wishes(WishID),
    PRIMARY KEY (UserID, WishID)
);
